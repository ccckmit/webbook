jQuery 的 each 函數

範例程式

<html>
 <head>
  <SCRIPT language="JavaScript" SRC="jquery.js"></SCRIPT>
 </head>
 <body>
  <b id="bold1">bold</b><BR/>
  <a id="link1" href="http://www.yahoo.com/">Yahoo</a><BR/>
  <script type="text/javascript">
  $('*').each(function(i) {
    document.write("node "+i+":nodeName="+this.nodeName+" id="+this.id+"<br/>");
  });
  </script>
 </body>
</html>
